if ishero
{
	checkskillrank("neow_droparmy")
	if skillrank > 0
	{
		neow_summon_skill_drop = 1
	}
	checkskillrank("neow_supportarmy")
	if skillrank > 0
	{
		neow_summon_skill_support = 1
	}
}

skillrankbp=checkskillrank("neow_boostpower")
skillranksd=checkskillrank("neow_supportdrop")
if (skillrankbp)&(skillranksd)
{
    global_read("armiesenter")
    array_read("result",1-left)
    army = result
    yourarmycost = armyvalue(1)
    array_read("result",left)
    army = result
    oppositearmycost = armyvalue(1)
    diff=((oppositearmycost-yourarmycost)/yourarmycost)

    diff+=0.01*(skillrankbp+skillranksd)

    if diff > 1
    {
        diff=1
    }
    if diff < 0
    {
        diff=0
    }
    if skillrankbp
    {
      mhealth*=(1+diff)
      mdamage*=(1+diff)
    }
    if ishero
    {
        if skillranksd
        {
            neow_summon_skill_supportDrop = diff
        }
    }
}
