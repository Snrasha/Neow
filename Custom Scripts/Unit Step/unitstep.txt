	{
	neow_summon_skill_support = 0
	checkskillrank("neow_supportarmy")
	if skillrank > 0
	{
		pickunitfromgroup("neow_support_ranged",skillrank/5)
		summonunit(foundunit,floor(random(skillrank*2)))
		pickunitfromgroup("neow_support_melee",skillrank/5)
		summonunit(foundunit,skillrank*3)
	}
}
if (ishero)&(neow_summon_skill_drop)
	{
	neow_summon_skill_drop = 0
	checkskillrank("neow_droparmy")
	if skillrank > 0
	{
		pickunitfromgroup("neow_support_melee",skillrank/5)
		summonunit(foundunit,floor(random(skillrank*2)))
		pickunitfromgroup("neow_support_ranged",skillrank/5)
		summonunit(foundunit,skillrank*3)
	}
}
if (ishero)&(neow_summon_skill_supportDrop)
{
	checkskillrank("neow_supportdrop")
	if skillrank > 0
	{
		range=0
		if(skillrank > 2){
		 range=1
		}
		pickunitfromgroup("neow_support_melee",range)
		summonunit(foundunit,floor(neow_summon_skill_supportDrop*(6+skillrank)))
		pickunitfromgroup("neow_support_ranged",range)
		summonunit(foundunit,floor(neow_summon_skill_supportDrop*(6+skillrank)))
	}
